## Web Quickstart

<Steps>
  <Step title="Install the SDK">
    ```bash
    npm install @stringboot/web-sdk
    ```
  </Step>
  <Step title="Initialize">
    ```javascript
    import StringBoot from '@stringboot/web-sdk';
    
    await StringBoot.initialize({
      apiToken: 'your-api-token-here',
      baseUrl: 'https://api.stringboot.com',
      defaultLanguage: 'en',
      debug: true
    });
    ```
  </Step>
  <Step title="Use strings">
    **Vanilla JavaScript:**

    ```javascript
    // Get a string
    const welcome = await StringBoot.get('welcome_message');
    
    // Watch for changes
    StringBoot.watch('welcome_message', (value) => {
      document.getElementById('welcome').textContent = value;
    });
    
    // Change language
    await StringBoot.changeLanguage('es');
    ```

    **React:**

    ```jsx
    import { useString, useLanguage } from '@stringboot/web-sdk/react';
    
    function App() {
      const title = useString('app_title');
      const [currentLang, setLanguage] = useLanguage();
    
      return (
        <div>
          <h1>{title}</h1>
          <button onClick={() => setLanguage('es')}>Español</button>
        </div>
      );
    }
    ```
  </Step>
</Steps>

## Android Quickstart

<Steps>
  <Step title="Add dependency">
    Add to your `build.gradle.kts`:

    ```kotlin
    dependencies {
        implementation("com.stringboot:stringboot-android-sdk:1.0.8")
    }
    ```
  </Step>
  <Step title="Configure AndroidManifest.xml">
    ```xml
    <manifest xmlns:android="http://schemas.android.com/apk/res/android">
        <uses-permission android:name="android.permission.INTERNET" />
    
        <application android:name=".StringbootApplication">
            <meta-data
                android:name="com.stringboot.api_url"
                android:value="https://api.stringboot.com" />
            <meta-data
                android:name="com.stringboot.api_token"
                android:value="YOUR_API_TOKEN_HERE" />
        </application>
    </manifest>
    ```
  </Step>
  <Step title="Initialize in Application class">
    ```kotlin
    class StringbootApplication : Application() {
        override fun onCreate() {
            super.onCreate()
            StringbootExtensions.autoInitialize(this)
        }
    }
    ```
  </Step>
  <Step title="Use strings">
    **XML with tags:**

    ```xml
    <TextView
        android:tag="welcome_message"
        android:text="@string/welcome_message" />
    ```

    **In Activity:**

    ```kotlin
    binding.root.applyStringbootTags()
    
    // Or get programmatically
    val text = StringProvider.get("welcome_message")
    ```
  </Step>
</Steps>

## iOS Quickstart

<Steps>
  <Step title="Add Swift Package">
    In Xcode: **File → Add Packages**

    ```
    https://github.com/stringboot/stringboot-ios-sdk.git
    ```
  </Step>
  <Step title="Initialize in App">
    ```swift
    import StringbootSDK
    
    @main
    struct MyApp: App {
        init() {
            StringProvider.shared.initialize(
                cacheSize: 1000,
                apiToken: "your-api-token-here",
                baseURL: "https://api.stringboot.com",
                autoSync: true
            )
        }
    
        var body: some Scene {
            WindowGroup {
                ContentView()
            }
        }
    }
    ```
  </Step>
  <Step title="Use strings">
    ```swift
    import SwiftUI
    import StringbootSDK
    
    struct ContentView: View {
        @ObservedObject var stringProvider = StringProvider.shared
    
        var body: some View {
            VStack {
                // Auto-updating text
                SBText("welcome_message")
                    .font(.title)
    
                // Language picker
                Picker("Language", selection: $selectedLanguage) {
                    ForEach(stringProvider.availableLanguages, id: \.code) { lang in
                        Text(lang.name).tag(lang.code)
                    }
                }
                .onChange(of: selectedLanguage) { _, newLang in
                    Task {
                        await stringProvider.changeLanguage(to: newLang)
                    }
                }
            }
        }
    }
    ```
  </Step>
</Steps>
